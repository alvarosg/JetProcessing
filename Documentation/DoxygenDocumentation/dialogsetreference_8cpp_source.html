481 void DialogSetReference::SetControlsStep(int newStep){
482  int i;
483  int nSteps=ui->tabWidget->count();

485  if(newStep<0 || newStep >= nSteps)
486  return;
487 
488  step=newStep;

491  ui->tabWidget->setTabEnabled(i,false);

495

501  ui->buttonPrevious->setEnabled(true);
502  ui->buttonNext->setEnabled(false);
503  }

505  ui->buttonPrevious->setEnabled(true);
