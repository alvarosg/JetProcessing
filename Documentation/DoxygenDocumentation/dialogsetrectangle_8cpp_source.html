296 void DialogSetRectangle::MouseLeftButtonMoved(int x,int y){
297  //Téngase en cuenta que no hace falta redibujar nada, ya que al cambiar los valores, automáticamente se actualiza la ventana
298 
299  //Si es un rectángulo
300  if(rectRect!=NULL){
301  if (x<ui->spinXMin->value())
302  ui->spinXMin->setValue(x);
303  if (y<ui->spinYMin->value())
304  ui->spinYMin->setValue(y);

306  ui->spinYMax->setValue(y);
307  }
308  //Si es una recta
309  else{
310  switch(pointBeingMoved){
311  //Si he pinchado donde no hay nada
312  case 0:
313  ui->spinXMax->setValue(x);
314  ui->spinYMax->setValue(y);
315  break;
316  //Si he pinchado en el primer punto
317  case 1:
318  ui->spinXMin->setValue(x);
319  ui->spinYMin->setValue(y);
320  break;
321  //Si he pinchado en el segundo punto
322  case 2:
323  ui->spinXMax->setValue(x);
324  ui->spinYMax->setValue(y);
325  break;
326  //Si he pinchado en el centro
327  case 3:
328  QRect rectAux=*rect;
329  rectAux.setLeft(x-(rect->right()-rect->left())/2);
330  rectAux.setRight(x+(rect->right()-rect->left())/2);
331  rectAux.setTop(y-(rect->bottom()-rect->top())/2);
332  rectAux.setBottom(y+(rect->bottom()-rect->top())/2);
333  ui->spinXMin->setValue(rectAux.left());
334  ui->spinXMax->setValue(rectAux.right());
335  ui->spinYMin->setValue(rectAux.top());
336  ui->spinYMax->setValue(rectAux.bottom());
337  break;
338 
339  }
