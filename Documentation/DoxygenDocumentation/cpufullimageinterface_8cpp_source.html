137 void FFT(float *real, float *imag, int size, int dir){
138  int npoints;
139  // Number of points needed
140  int times=0;
141  for (npoints=1;npoints<size;times++)
142  npoints *= 2;
143 
144  // Reversal
145  int i2 = npoints >> 1;
146  int j = 0;
147  int k;
148  for (int i=0;i<npoints-1;i++) {
149  if (i < j) {
150  float x = real[i];
151  float y = imag[i];
152  real[i] = real[j];
153  imag[i] = imag[j];
154  real[j] = x;
155  imag[j] = y;
156  }
157  k = i2;
158  while (k <= j) {
159  j -= k;
160  k >>= 1;
161  }
162  j += k;
163  }
164 
165  //Compute the FFT
166  float c1 = -1.0;
167  float c2 = 0.0;
168  int l2 = 1;
169  for (int l=0;l<times;l++) {
170  int l1 = l2;
171  l2 <<= 1;
172  float u1 = 1.0;
173  float u2 = 0.0;
174  for (int j=0;j<l1;j++) {
175  for (int i=j;i<npoints;i+=l2) {
176  int i1 = i + l1;
177  float t1 = u1 * real[i1] - u2 * imag[i1];
178  float t2 = u1 * imag[i1] + u2 * real[i1];
179  real[i1] = real[i] - t1;
180  imag[i1] = imag[i] - t2;
181  real[i] += t1;
182  imag[i] += t2;
183  }
184  float z = u1 * c1 - u2 * c2;
185  u2 = u1 * c2 + u2 * c1;
186  u1 = z;
187  }
188  c2 = (float) sqrt((1.0 - c1) / 2.0);
189  if (dir == 1)
190  c2 = -c2;
191  c1 = (float) sqrt((1.0 + c1) / 2.0);
192  }
193 
194  //Scaling on the inverse
195  if (dir == INVERSE) {
196  for (int i=0;i<npoints;i++) {
197  real[i] = real[i]/npoints;
198  imag[i] = imag[i]/npoints;
199  }
200  }
